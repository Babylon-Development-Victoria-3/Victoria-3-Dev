namespace = infamy_rebalance
infamy_rebalance.1 = {
	type = country_event
	hidden = yes
	trigger = {
		OR = {
			AND = {
				infamy >= infamy_threshold:pariah
				any_state = {
					owner = ROOT
					OR = {
						has_modifier = modifier_infamy_infamous
						has_modifier = modifier_infamy_notorious
						NOT = {
							has_modifier = modifier_infamy_pariah
						}
					}
				}
			}
			AND = {
				infamy >= infamy_threshold:notorious
				NOT = {
					infamy >= infamy_threshold:pariah
				}
				any_state = {
					owner = ROOT
					OR = {
						has_modifier = modifier_infamy_infamous
						NOT = {
							has_modifier = modifier_infamy_notorious
						}
						has_modifier = modifier_infamy_pariah
					}
				}
			}
			AND = {
				infamy >= infamy_threshold:infamous
				NOT = {
					infamy >= infamy_threshold:notorious
				}
				any_state = {
					owner = ROOT
					OR = {
						NOT = {
							has_modifier = modifier_infamy_infamous
						}
						has_modifier = modifier_infamy_notorious
						has_modifier = modifier_infamy_pariah
					}
				}
			}
			AND = {
				NOT = {
					infamy >= infamy_threshold:infamous
				}
				any_state = {
					owner = ROOT
					OR = {
						has_modifier = modifier_infamy_infamous
						has_modifier = modifier_infamy_notorious
						has_modifier = modifier_infamy_pariah
					}
				}
			}
		}
	}
	immediate = {
		#name = infamy_rebalance.1.a
		if = {
			limit = {
				infamy >= infamy_threshold:pariah
			}
			every_scope_state = {
				limit = {
					OR = {
						has_modifier = modifier_infamy_infamous
						has_modifier = modifier_infamy_notorious
						NOT = {
							OR = {
								has_modifier = modifier_infamy_infamous
								has_modifier = modifier_infamy_notorious
							}
						}
					}
				}
				remove_modifier = modifier_infamy_infamous
				remove_modifier = modifier_infamy_notorious
				remove_modifier = modifier_infamy_pariah
				add_modifier = {
					name = modifier_infamy_pariah
				}
			}
			every_country = {
				limit = {
					AND = {
						has_interest_marker_in_region = root.capital.region
						NOT = {
							this = root
						}
					}
				}
				add_journal_entry = {
					type = je_pariah_neighbour
					target = root
				}
			}
		}
		else_if = {
			limit = {
				infamy >= infamy_threshold:notorious
			}
			every_scope_state = {
				limit = {
					OR = {
						has_modifier = modifier_infamy_infamous
						has_modifier = modifier_infamy_pariah
						NOT = {
							OR = {
								has_modifier = modifier_infamy_infamous
								has_modifier = modifier_infamy_pariah
							}
						}
					}
				}
				remove_modifier = modifier_infamy_infamous
				remove_modifier = modifier_infamy_notorious
				remove_modifier = modifier_infamy_pariah
				add_modifier = {
					name = modifier_infamy_notorious
				}
			}
		}
		else_if = {
			limit = {
				infamy >= infamy_threshold:infamous
			}
			every_scope_state = {
				limit = {
					OR = {
						has_modifier = modifier_infamy_notorious
						has_modifier = modifier_infamy_pariah
						NOT = {
							OR = {
								has_modifier = modifier_infamy_notorious
								has_modifier = modifier_infamy_pariah
							}
						}
					}
				}
				remove_modifier = modifier_infamy_infamous
				remove_modifier = modifier_infamy_notorious
				remove_modifier = modifier_infamy_pariah
				add_modifier = {
					name = modifier_infamy_infamous
				}
			}
		}
		else = {
			every_scope_state = {
				limit = {
					OR = {
						has_modifier = modifier_infamy_infamous
						has_modifier = modifier_infamy_notorious
						has_modifier = modifier_infamy_pariah
					}
				}
				remove_modifier = modifier_infamy_infamous
				remove_modifier = modifier_infamy_notorious
				remove_modifier = modifier_infamy_pariah
			}
		}
	}
}

infamy_rebalance.2 = {
	type = country_event
	hidden = yes
	immediate = {
		change_infamy = 25
	}
}

infamy_rebalance.3 = {
	type = country_event
	hidden = yes
	immediate = {
		change_infamy = -25
	}
}
