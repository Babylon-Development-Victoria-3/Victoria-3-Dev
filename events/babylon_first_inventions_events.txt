namespace = first_inventions_events
first_inventions_events.military_aviation = {
	#Airplanes discovered by X
	type = country_event
	event_image = {
		video = "gfx/event_pictures/unspecific_airplane.bk2"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/airplane"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	title = first_inventions_events.military_aviation.t
	desc = first_inventions_events.military_aviation.d
	flavor = first_inventions_events.military_aviation.f
	duration = 3
	immediate = {
		random_country = {
			limit = {
				has_modifier = military_aviation_invented_country
			}
			save_scope_as = inventor_scope
		}
	}
	option = {
		trigger = {
			can_research = military_aviation
		}
		name = first_inventions_events.military_aviation.close
	}
	option = {
		trigger = {
			NOT = {
				can_research = military_aviation
			}
		}
		name = first_inventions_events.military_aviation.far_off
	}
}

first_inventions_events.skyscrapers = {
	#First skyscraper built by X
	type = country_event
	event_image = {
		video = "gfx/event_pictures/unspecific_world_fair.bk2"
	}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/world_fair"
	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"
	title = first_inventions_events.skyscrapers.t
	desc = first_inventions_events.skyscrapers.d
	flavor = first_inventions_events.skyscrapers.f
	duration = 3
	immediate = {
		random_country = {
			limit = {
				#This modifier is from the vanilla skyscraper event. Being the first to build one gives this modifier for a few years.
				has_modifier = first_skyscraper_modifier
			}
			save_scope_as = inventor_scope
		}
	}
	option = {
		trigger = {
			NOT = {
				has_technology_researched = steel_frame_buildings
				has_technology_researched = reinforced_concrete
			}
		}
		name = first_inventions_events.skyscrapers.far_off
	}
	option = {
		trigger = {
			has_technology_researched = steel_frame_buildings
			has_technology_researched = reinforced_concrete
		}
		name = first_inventions_events.skyscrapers.close
	}
}
