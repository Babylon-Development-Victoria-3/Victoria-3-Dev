	}

is_congress_target = {
	OR = {
		this = global_var:babylon_vienna_congress_session_target
		is_target_in_global_variable_list = {
			name = babylon_vienna_congress_session_target_state_owners
			target = this
		}
	}
	this != global_var:babylon_vienna_congress_session_initiator
}

is_congress_attendee = {
	OR = {
		is_congress_supporter = yes
		is_congress_abstainer = yes
		is_congress_opposer = yes
		is_congress_observer = yes
	}
}
