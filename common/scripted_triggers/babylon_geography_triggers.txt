#Checks if this and TARGET are both from the same continent. 
#Theoretically.
babylon_is_same_continent = {
	OR = {
		#Check if both are in europe
		AND = {
			country_is_in_europe = yes
			$TARGET$ = {
				country_is_in_europe = yes
			}
		}
		#Are in North + Central America
		AND = {
			OR = {
				country_is_in_north_america = yes
				country_is_in_central_america = yes
			}
			$TARGET$ = {
				OR = {
					country_is_in_north_america = yes
					country_is_in_central_america = yes
				}
			}
		}
		#Are both South American?
		AND = {
			country_is_in_south_america = yes
			$TARGET$ = {
				country_is_in_south_america = yes
			}
		}
		#African?
		AND = {
			country_is_in_africa = yes
			$TARGET$ = {
				country_is_in_africa = yes
			}
		}
		#Pan-asian?
		AND = {
			OR = {
				country_is_in_middle_east = yes
				country_is_in_central_asia = yes
				country_is_in_india = yes
				country_is_in_east_asia = yes
				country_is_in_southeast_asia = yes
			}
			$TARGET$ = {
				OR = {
					country_is_in_middle_east = yes
					country_is_in_central_asia = yes
					country_is_in_india = yes
					country_is_in_east_asia = yes
					country_is_in_southeast_asia = yes
				}
			}
		}
	}
}
