babylon_calc_cohesion_governance_principles = {
	if = {
		limit = {
			has_law = law_type:law_chiefdom
		}
		if = {
			limit = {
				$TARGET$ = {
					OR = {
						has_law = law_type:law_chiefdom
						has_law = law_type:law_monarchy
					}
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					OR = {
						has_law = law_type:law_presidential_republic
						has_law = law_type:law_theocracy
					}
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 1
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_parliamentary_republic
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 2
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 15
			}
		}
	}
	#Exit chiefdom
	else_if = {
		limit = {
			has_law = law_type:law_monarchy
		}
		if = {
			limit = {
				$TARGET$ = {
					OR = {
						has_law = law_type:law_theocracy
						has_law = law_type:law_monarchy
					}
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_presidential_republic
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 1
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_parliamentary_republic
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 2
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 20
			}
		}
	}
	#Exit monarchy
	else_if = {
		limit = {
			has_law = law_type:law_presidential_republic
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_presidential_republic
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_parliamentary_republic
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 1
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_theocracy
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 2
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 10
			}
		}
	}
	#Exit presidential republic
	else_if = {
		limit = {
			has_law = law_type:law_parliamentary_republic
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_parliamentary_republic
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_theocracy
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 3
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 4
			}
		}
	}
	#Exit presidential republic
	else_if = {
		limit = {
			has_law = law_type:law_theocracy
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_theocracy
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			change_local_variable = {
				name = cohesion_change
				add = 15
			}
		}
	}
	#Exit theocracy
	else_if = {
		limit = {
			has_law = law_type:law_council_republic
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			#No change
		}
	}
}
