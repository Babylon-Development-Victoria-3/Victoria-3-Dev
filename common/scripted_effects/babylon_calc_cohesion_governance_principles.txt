babylon_calc_discord_governance_principles = {
	if = {
		limit = {
			root = {
				has_law = law_type:law_chiefdom
			}
		}
		if = {
			limit = {
				$TARGET$ = {
					OR = {
						has_law = law_type:law_chiefdom
						has_law = law_type:law_monarchy
					}
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					OR = {
						has_law = law_type:law_presidential_republic
						has_law = law_type:law_theocracy
					}
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_miniscule
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_parliamentary_republic
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_small
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_massive
			}
		}
	}
	#Exit chiefdom
	else_if = {
		limit = {
			root = {
				has_law = law_type:law_monarchy
			}
		}
		if = {
			limit = {
				$TARGET$ = {
					OR = {
						has_law = law_type:law_theocracy
						has_law = law_type:law_monarchy
					}
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_presidential_republic
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_miniscule
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_parliamentary_republic
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_small
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_massive
			}
		}
	}
	#Exit monarchy
	else_if = {
		limit = {
			root = {
				has_law = law_type:law_presidential_republic
			}
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_presidential_republic
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_parliamentary_republic
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_miniscule
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_theocracy
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_small
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_huge
			}
		}
	}
	#Exit presidential republic
	else_if = {
		limit = {
			root = {
				has_law = law_type:law_parliamentary_republic
			}
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_parliamentary_republic
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_theocracy
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_medium
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_large
			}
		}
	}
	#Exit presidential republic
	else_if = {
		limit = {
			root = {
				has_law = law_type:law_theocracy
			}
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_theocracy
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_massive
			}
		}
	}
	#Exit theocracy
	else_if = {
		limit = {
			root = {
				has_law = law_type:law_council_republic
			}
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_council_republic
				}
			}
			#No change
		}
	}
}
