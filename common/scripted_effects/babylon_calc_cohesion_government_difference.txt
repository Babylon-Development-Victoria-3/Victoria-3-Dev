babylon_calc_cohesion_government_difference = {
	#Preliminary setup
	$TARGET$ = {
		save_scope_as = target_country
	}
	set_local_variable = {
		name = cohesion_change
		value = 0
	}
	########################################
	### INSERT CALCULATION EFFECTS BELOW ###
	babylon_calc_cohesion_distribution_of_power = {
		TARGET = scope:target_country
	}
	babylon_calc_cohesion_governance_principles = {
		TARGET = scope:target_country
	}
	babylon_calc_cohesion_diplomatic_stance = {
		TARGET = scope:target_country
	}
	### INSERT CALCULATION EFFECTS ABOVE ###
	########################################
	#Apply the change to both self (and target?)
	if = {
		limit = {
			$IS_GAIN$ = yes
		}
		change_variable = {
			name = theoretical_cohesion
			add = local_var:cohesion_change
		}
		scope:target_country = {
			change_variable = {
				name = theoretical_cohesion
				add = root.local_var:cohesion_change
			}
		}
	}
	else_if = {
		limit = {
			$IS_GAIN$ = no
		}
		change_variable = {
			name = political_cohesion
			subtract = local_var:cohesion_change
		}
		scope:target_country = {
			change_variable = {
				name = political_cohesion
				subtract = root.local_var:cohesion_change
			}
		}
	}
}
