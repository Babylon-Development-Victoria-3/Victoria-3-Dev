babylon_discord_fetch = {
	$TARGET_COUNTRY$ = {
		save_scope_as = target_country
	}
	set_local_variable = {
		name = discord_fetch_result
		value = 0
	}
	if = {
		limit = {
			has_diplomatic_pact = {
				who = root
				type = defensive_pact
			}
		}
		change_local_variable = {
			name = discord_fetch_result
			add = {
				value = babylon_discord_external_calculation
				multiply = political_discord_defensive_pact_mult
			}
		}
	}
	if = {
		limit = {
			has_diplomatic_pact = {
				who = root
				type = alliance
			}
		}
		change_local_variable = {
			name = discord_fetch_result
			add = {
				value = babylon_discord_external_calculation
				multiply = political_discord_alliance_pact_mult
			}
		}
	}
	if = {
		limit = {
			has_diplomatic_pact = {
				who = root
				type = trade_agreement
			}
		}
		change_local_variable = {
			name = discord_fetch_result
			add = {
				value = babylon_discord_external_calculation
				multiply = political_discord_trade_pact_mult
			}
		}
	}
	if = {
		limit = {
			has_diplomatic_pact = {
				who = root
				type = customs_union
			}
		}
		change_local_variable = {
			name = discord_fetch_result
			add = {
				value = babylon_discord_external_calculation
				multiply = political_discord_customs_union_pact_mult
			}
		}
	}
	if = {
		limit = {
			has_diplomatic_pact = {
				who = root
				type = rivalry
			}
		}
		change_local_variable = {
			name = discord_fetch_result
			subtract = {
				value = babylon_discord_external_calculation
				multiply = political_discord_rivalry_pact_mult
			}
		}
	}
}
