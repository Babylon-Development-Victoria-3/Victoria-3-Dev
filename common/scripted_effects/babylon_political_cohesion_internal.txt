change_cohesion = {
	# $VALUE$ = num [should be the value that political cohesion changes by temporarily, before being adjusted by the monthly pulse gradually]
	save_temporary_scope_value_as = {
		name = cohesion_change
		value = $VALUE$
	}
	if = {
		limit = { 
			$VALUE$ > 0
		}
		custom_tooltip = increase_cohesion_tt
	}
	else_if = {
		limit = { 
			$VALUE$ < 0
		}
		custom_tooltip = decrease_cohesion_tt
	}
	change_variable = {
		name = political_cohesion
		add = $VALUE$
	}
	clamp_variable = {
		name = political_cohesion
		max = 400
		min = 0
	}
}
