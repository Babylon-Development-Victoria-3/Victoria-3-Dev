babylon_calc_discord_CHARACTER_ideology = {
	set_local_variable = {
		name = babylon_discord_character_ideology
		value = 0
	}
	every_law = {
		type = {
			save_temporary_scope_as = law_type_scope
			babylon_calc_LAW_stance = {
				LAW = scope:law_type_scope
				CHARACTER = $CHARACTER$
			}
		}
	}
	change_local_variable = {
		name = babylon_discord_character_ideology
		multiply = $MULTIPLIER$
	}
	change_variable = {
		name = theoretical_discord
		add = local_var:babylon_discord_character_ideology
	}
	remove_local_variable = babylon_discord_character_ideology
}

babylon_calc_LAW_stance = {
	$CHARACTER$ = {
		if = {
			limit = {
				law_stance = {
					law = $LAW$
					value = strongly_disapprove
				}
			}
			change_local_variable = {
				name = babylon_discord_character_ideology
				add = political_discord_change_small
			}
		}
		else_if = {
			limit = {
				law_stance = {
					law = $LAW$
					value = disapprove
				}
			}
			change_local_variable = {
				name = babylon_discord_character_ideology
				add = political_discord_change_miniscule
			}
		}
		else_if = {
			limit = {
				law_stance = {
					law = $LAW$
					value = approve
				}
			}
			change_local_variable = {
				name = babylon_discord_character_ideology
				subtract = political_discord_change_miniscule
			}
		}
		else_if = {
			limit = {
				law_stance = {
					law = $LAW$
					value = strongly_approve
				}
			}
			change_local_variable = {
				name = babylon_discord_character_ideology
				subtract = political_discord_change_small
			}
		}
	}
}
