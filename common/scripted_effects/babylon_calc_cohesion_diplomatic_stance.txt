babylon_calc_discord_diplomatic_stance = {
	if = {
		limit = {
			root = {
				has_law = law_type:law_expansionism
			}
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_expansionism
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_imperialism
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_miniscule
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					OR = {
						has_law = law_type:law_armed_neutrality
						has_law = law_type:law_pacifist
					}
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_small
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					OR = {
						has_law = law_type:law_anti_imperialist
						has_law = law_type:law_true_neutrality
					}
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_xl
			}
		}
	}
	#Exit expansionism
	else_if = {
		limit = {
			root = {
				has_law = law_type:law_imperialism
			}
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_imperialism
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_armed_neutrality
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_small
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_pacifist
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_medium
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					OR = {
						has_law = law_type:law_anti_imperialist
						has_law = law_type:law_true_neutrality
					}
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_xl
			}
		}
	}
	#Exit imperialism
	else_if = {
		limit = {
			root = {
				has_law = law_type:law_armed_neutrality
			}
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_armed_neutrality
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					OR = {
						has_law = law_type:law_anti_imperialist
						has_law = law_type:law_pacifist
					}
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_miniscule
			}
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_true_neutrality
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_medium
			}
		}
	}
	#Exit anti imperialist
	else_if = {
		limit = {
			root = {
				has_law = law_type:law_pacifist
			}
		}
		if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_pacifist
				}
			}
			#No change
		}
		else_if = {
			limit = {
				$TARGET$ = {
					has_law = law_type:law_true_neutrality
				}
			}
			change_discord_factor = {
				VALUE = political_discord_change_small
			}
		}
	}
	#Exit pacifist
	#True neutral gives no change to itself
}
