babylon_quick_enact_law = {
	scope = law
	saved_scopes = {
		country
	}

	effect = {
		scope:country = {
			if = {
				limit = {
					NOT = {
						has_law = prev.type
					}
				}
				start_enactment = prev.type
				add_enactment_phase = 2 # is capped so just to make sure it remains functional if paradox changes anything
				add_law_progress = 0.9
				add_enactment_modifier = {
					name = law_enactment_chance
				}				#activate_law = prev.type
			}
		}
	}

	is_valid = {
		scope:country = {
			NOT = {
				has_law = prev.type
			}
		}

		trigger_if = {
			limit = {
				type = law_type:law_anti_imperialist
			}
			scope:country = {
				has_law != law_colonial_exploitation
				has_law != law_colonial_resettlement	
			}
		}

		trigger_if = {
			limit = {
				type = law_type:law_pacifist
			}
			scope:country = {
				has_law != law_professional_army
				has_law != law_national_guard
				has_law != law_militarized_police
			}
		}
	}


	is_shown = {
		type != law_type:law_true_neutrality
	}
}
