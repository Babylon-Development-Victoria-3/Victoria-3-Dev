babylon_quick_enact_law = {
	scope = law
	saved_scopes = {
		country
	}

	effect = {
		scope:country = {
			if = {
				limit = {
					NOT = {
						has_law = prev.type
					}
				}
				activate_law = prev.type
			}
		}
	}

	is_shown = {
		scope:country = {
			NOT = {
				has_law = prev.type
			}
		}

	}
}
