je_frankfurt_conference_vote_republic_button = {
	name = "je_frankfurt_conference_vote_republic_button"
	desc = "je_frankfurt_conference_vote_republic_button_desc"
	visible = {
		var:frankfurt_conference_current_vote = 1
	}
	possible = {
		custom_tooltip = {
			text = frankfurt_conference_voting_governance_principle
			var:frankfurt_conference_current_vote = 1
		}
		hidden_trigger = {
			NOT = {
				has_variable = frankfurt_conference_voted_republic
			}
		}
	}
	selected = {
		has_variable = frankfurt_conference_voted_republic
	}
	effect = {
		set_variable = frankfurt_conference_voted_republic
		## Remove this voting var (if it exists, thats what the ?= means)
		remove_variable ?= frankfurt_conference_voted_monarchy
		custom_tooltip = {
			text = frankfurt_votes_for_republic_tt
		}
	}
}
