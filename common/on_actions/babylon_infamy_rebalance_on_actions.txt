on_monthly_pulse_country = {
	#updates provinces each month
	events = {
		infamy_rebalance.1
	}
	random_events = {
		#Fires the pariah wartime event roughly once every 4 months. 
		3 = 0
		1 = infamy_rebalance.8 #wartime event for pariah
	}
}
# Root = Diplomatic Play
on_diplo_play_war_start = {
	effect = {
		if = {
			limit = {
				has_play_goal = contain_threat
			}
			every_country = {
				limit = {
					has_variable = pariah_coalition_member
					has_strategic_adjacency = scope:root.target
					NOT = {
						is_diplomatic_play_enemy_of = scope:root.target
					}
				}
				add_modifier = {
					name = scared_of_pariah_prestige_loss
					months = normal_modifier_time
					is_decaying = yes
				}
			}
		}
	}
}
