# Root = The applicable country
# scope:technology = The technology type they just acquired
on_babylon_acquired_technology = {
	effect = {
		if = {
			limit = { 
				#if the tech is broadcastable
				scope:technology = {
					OR = {
						this = military_aviation
						this = combustion_engine
						this = telephone
						this = electrical_generation
						this = radio
						this = dreadnought
						this = mobile_armor
						this = pumpjacks
					}
				}
			}
			#Then broadcast the tech
			scope:technology = {
				babylon_aquire_TECH_and_broadcast ={
					TECH = this
				}
			}
		}
	}
}
