
# Root = The applicable country
# scope:technology = The technology type they just acquired
on_acquired_technology = {
	effect = {
		if = {
			limit = {
				has_technology_researched = military_aviation
				NOT = {
					has_global_variable = military_aviation_invented_global
				}
			 }
			 set_global_variable = military_aviation_invented_global
			 add_modifier = {
				name = military_aviation_invented_country
			 }
			 every_country = {
				limit = {
					NOT = {
						this = root
					}
				}
				#delay event by 1 day. Because PDX.
				trigger_event = {
					id = first_inventions_events.aviation
					popup = yes
					days = 1
				}
			 }
		}
	}
}
