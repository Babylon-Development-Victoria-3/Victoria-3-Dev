je_autocracy = {
	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	is_shown_when_inactive = {
		OR = {
			has_law = law_type:law_monarchy
			has_law = law_type:law_theocracy
		}

		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_oligarchy
		}

		OR = {
			has_law = law_type:law_state_religion
			has_law = law_type:law_jizya
			has_law = law:type:law_denominational_exclusion
			#^^The above two come from VFM, can be commented out at anyones convenience
		}
	}

	possible = {
		OR = {
			ig:ig_landowners = {
				is_powerful = yes
			}
			ig:ig_devout = {
				is_powerful = yes
			}
		}
	}

	immediate = {
		set_variable = {
			name = autocracy_var
			value = 1
		}
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = {
					OR = {
						has_law = law_type:law_monarchy
						has_law = law_type:law_theocracy
					}

					OR = {
						has_law = law_type:law_autocracy
						has_law = law_type:law_oligarchy
					}

					OR = {
						has_law = law_type:law_state_religion
						#has_law = law_type:law_jizya
						#has_law = law:type:law_denominational_exclusion
						#^^The above two come from VFM, can be commented out at anyones convenience
					}

					NOT = {
						any_civil_war = {
							civil_war_progress > 0.5
						}
					}
				}
				change_variable = {
					name = autocracy_var
					add = 1
				}
			}
		}
		random_events = {
			10 = 0
			1 = autocracy.2 #Secret Societies event. Punish or let be an IG against authoritarianism - Intelligentsia only?
			1 = autocracy.3 #Blessed are the Meek. Extra loyalists/less radicals from SoL changes
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
		OR = {
			has_law = law_type:law_monarchy
			has_law = law_type:law_theocracy
		}
		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_oligarchy
		}
		OR = {
			has_law = law_type:law_state_religion
			has_law = law_type:law_jizya
			has_law = law:type:law_denominational_exclusion
			#^^The above two come from VFM, can be commented out at anyones convenience
		}		
		NOT = {
			any_civil_war = {
				civil_war_progress > 0.5
			}
		}
	}

	on_complete = {
		trigger_event = {
			id = autocracy.1 #Currently is 'Official Nationality' event 
		}
	}

	fail = {
		NOT = {
			OR = {
				has_law = law_type:law_monarchy
				has_law = law_type:law_theocracy
			}
	
			OR = {
				has_law = law_type:law_autocracy
				has_law = law_type:law_oligarchy
			}
	
			OR = {
				has_law = law_type:law_state_religion
				#has_law = law_type:law_jizya
				#has_law = law:type:law_denominational_exclusion
				#^^The above two come from VFM, can be commented out at anyones convenience
			}
		}
	}

	current_value = {
		value = root.var:autocracy_var
	}

	goal_add_value = {
		add = 360
	}

	progressbar = yes

	weight = 100
}
