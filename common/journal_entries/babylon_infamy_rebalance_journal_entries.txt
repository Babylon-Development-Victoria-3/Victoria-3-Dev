je_pariah_neighbour = {
	icon = "gfx/interface/icons/event_icons/event_skull.dds"
	immediate = {
		add_modifier = {
			name = modifier_pariah_neighbour
		}
	}
	complete = {
		OR = {
			custom_tooltip = {
				text = cut_pariah_down_tt
				subject = scope:target
				NOT = {
					scope:target = {
						infamy >= infamy_threshold:pariah
					}
				}
			}
			NOT = {
				has_interest_marker_in_region = scope:target.capital.region
			}
			has_truce_with = scope:target
		}
	}
	on_complete = {
		remove_modifier = modifier_pariah_neighbour
	}
	on_monthly_pulse = {
		random_events = {
			22 = 0
			1 = infamy_rebalance.4
			1 = infamy_rebalance.5
		}
	}
	fail = {
	}
	invalid = {
	}
	progressbar = no
	should_be_pinned_by_default = yes
}
