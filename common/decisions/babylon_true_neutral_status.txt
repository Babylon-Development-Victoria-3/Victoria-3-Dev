ask_for_true_neutral_status = {
	is_shown = {
		has_law = law_type:law_neutral
		NOT = {
			has_journal_entry = je_babylon_true_neutral_status
		}
	}
	possible = {
		#Cant have any infamy
		infamy = 0
		#Cant be in a war
		is_at_war = no
		#Cant have ANY rivals
		NOT = {
			any_rival_country = {
				exists = yes
			}
		}
	}
	when_taken = {
		#Initialise the current signers
		set_variable = {
			name = true_neutral_current_signers_var
			value = 0
		}
		#Initialise required signers
		set_variable = {
			name = true_neutral_required_signers_var
			value = 0
		}
		every_country = {
			limit = {
				#fetch great powers and neighbours
				OR = {
					country_rank >= rank_value:great_power
					is_adjacent = root
				}
			}
			#give them an event that notifies them
			trigger_event = {
				id = babylon_true_neutral_status.1
				days = 3
			}
			#Add a required signer to the true neutral applicant
			root = {
				change_variable = {
					name = true_neutral_required_signers_var
					add = 1
				}
				add_to_variable_list = {
					name = true_neutral_applicant_signee_list
					target = prev
				}
			}
		}
		add_journal_entry = {
			type = je_babylon_true_neutral_status
		}

	}
	ai_chance = {
		base = 0
	}
}
consider_true_neutral_status = {
	is_shown = {
		#Show this decision if theres any applicants for true neutral status
		has_variable_list = true_neutral_applicants_consideration
	}
	possible = {
		hidden_trigger = {
			variable_list_size = {
				name = true_neutral_applicants_consideration
				value >= 1
			}
		}
		is_at_war = no
	}
	when_taken = {
		trigger_event = { 
			id = babylon_true_neutral_status.2
			popup = yes
		}
	}
	ai_chance = {
		base = 1
		
	}
}
